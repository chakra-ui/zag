```html
<script setup>
  import * as collapsible from "@zag-js/collapsible"
  import { normalizeProps, useMachine } from "@zag-js/vue"
  import { computed } from "vue"

  const [state, send] = useMachine(collapsible.machine({ id: "1" }))

  const api = computed(() =>
    collapsible.connect(state.value, send, normalizeProps),
  )
</script>

<template>
  <div v-bind="api.rootProps">
    <button v-bind="api.triggerProps">Collapse Trigger</button>
    <div v-bind="api.contentProps">Collape Content</div>
  </div>
</template>
```
