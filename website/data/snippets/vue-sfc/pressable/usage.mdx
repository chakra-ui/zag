```md
<script setup>
import * as pressable from "@zag-js/pressable"
import { normalizeProps, useMachine } from "@zag-js/vue"
import { computed } from "vue"

const [state, send] = useMachine(pressable.machine({ id: "1" }))

const api = computed(() => pressable.connect(state.value, send, normalizeProps))
</script>

<template>
  <button v-bind="api.pressableProps">
    <span v-if="api.isPressed">Pressed!</span>
    <span v-else>Press Me</span>
  </button>
</template>
```
