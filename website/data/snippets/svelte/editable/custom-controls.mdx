```svelte
<script lang="ts">
  import * as editable from "@zag-js/editable"
  import { useMachine, normalizeProps } from "@zag-js/svelte"
  const [snapshot, send] = useMachine(editable.machine({ id: "1" }))

  const api = $derived(editable.connect(snapshot, send, normalizeProps))
</script>

<div {...api.rootProps}>
  <div {...api.areaProps}>
    <input {...api.inputProps} />
    <span {...api.previewProps} />
  </div>
  <div>
    {#if !api.editing}
      <button {...api.editTriggerProps}>Edit</button>
    {/if}
    {#if api.editing}
      <div>
        <button {...api.submitTriggerProps}>Save</button>
        <button {...api.cancelTriggerProps}>Cancel</button>
      </div>
    {/if}
  </div>
</div>
```
