```tsx
import * as pagination from "@zag-js/pagination"
import { normalizeProps, useMachine } from "@zag-js/vue"
import { defineComponent, h, Fragment, computed } from "vue"
import { data } from "./data"

export default defineComponent({
  name: "Pagination",
  setup() {
    const [state, send] = useMachine(
      pagination.machine({ id: "1", count: data.length }),
    )

    const apiRef = computed(() =>
      pagination.connect(state.value, send, normalizeProps),
    )
    return () => {
      const api = apiRef.value

      return (
        <div>
          {api.totalPages > 1 && (
            <nav {...api.rootProps}>
              <ul>
                <li>
                  <a href="#previous" {...api.prevPageTriggerProps}>
                    Previous <span class="visually-hidden">Page</span>
                  </a>
                </li>
                {api.pages.map((page, i) => {
                  if (page.type === "page")
                    return (
                      <li key={page.value}>
                        <a href={`#${page.value}`} {...api.getPageTriggerProps(page)}>
                          {page.value}
                        </a>
                      </li>
                    )
                  else
                    return (
                      <li key={`ellipsis-${i}`}>
                        <span {...api.getEllipsisProps({ index: i })}>
                          &#8230;
                        </span>
                      </li>
                    )
                })}
                <li>
                  <a href="#next" {...api.nextPageTriggerProps}>
                    Next <span class="visually-hidden">Page</span>
                  </a>
                </li>
              </ul>
            </nav>
          )}
        </div>
      )
    }
  },
})
```
