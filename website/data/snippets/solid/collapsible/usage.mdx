```tsx
import * as collapsible from "@zag-js/collapsible"
import { normalizeProps, useMachine } from "@zag-js/solid"
import { createMemo, createUniqueId } from "solid-js"

const data = ["@chakra-ui/ark-ui", "@chakra-ui/panda", "@chakra-ui/chakra-ui"]

function Collapsible() {
  const [state, send] = useMachine(
    collapsible.machine({ id: createUniqueId() }),
  )

  const api = createMemo(() => collapsible.connect(state, send, normalizeProps))

  return (
    <div {...api().rootProps}>
      <button {...api().triggerProps}>
        {api().isOpen ? "Collapse" : "Expand"}
      </button>
      <div {...api().contentProps}>
        <Index each={items}>{(item) => <div>{item()}</div>}</Index>
      </div>
    </div>
  )
}
```
