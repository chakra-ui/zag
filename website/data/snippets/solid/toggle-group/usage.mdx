```jsx
import { normalizeProps, useMachine } from "@zag-js/solid"
import * as toggle from "@zag-js/toggle-group"
import { createMemo, createUniqueId } from "solid-js"

export function ToggleGroup() {
  const [state, send] = useMachine(toggle.machine({ id: createUniqueId() }))
  const api = createMemo(() => toggle.connect(state, send, normalizeProps))

  return (
    <div {...api().rootProps}>
      <button {...api().getItemProps({ value: "bold" })}>B</button>
      <button {...api().getItemProps({ value: "italic" })}>I</button>
      <button {...api().getItemProps({ value: "underline" })}>U</button>
    </div>
  )
}
```
